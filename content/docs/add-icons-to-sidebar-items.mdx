export const source = loader({
  // it assigns a URL to your pages
  baseUrl: '/docs',
  icon(icon) {  
    if (icon && icon in icons)
      return createElement(icons[icon as keyof typeof icons]);
  },
  source: docs.toFumadocsSource(),
});
